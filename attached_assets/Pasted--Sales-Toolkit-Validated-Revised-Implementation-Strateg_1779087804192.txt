### Persistence Hook Pattern (Corrected)
```typescript
// hooks/usePersistedSlice.ts - Generic hook for existing state slices
export const usePersistedSlice = <T>(
  sliceKey: string, // e.g., 'opportunities', 'drafts'
  initialState: T
) => {
  const [state, setState] = useState<T>(initialState);
  const [isHydrated, setIsHydrated] = useState(false);

  // Hydrate from IndexedDB on mount
  useEffect(() => {
    let mounted = true;
    get<T>(sliceKey).then((stored) => {
      if (mounted && stored) {
        setState(stored);
      }
      setIsHydrated(true);
    });
    return () => { mounted = false; };
  }, [sliceKey]);

  // Persist to IndexedDB when state changes (debounced)
  useEffect(() => {
    if (!isHydrated) return;
    
    const timeout = setTimeout(() => {
      set(sliceKey, state).catch((err) => {
        console.error(`Failed to persist ${sliceKey}`, err);
        // Could dispatch error to UI context here
      });
      
      // Broadcast to other tabs
      if (typeof BroadcastChannel !== 'undefined') {
        const channel = new BroadcastChannel('fieldkit_sync');
        channel.postMessage({ type: 'SLICE_UPDATED', sliceKey });
        channel.close();
      }
    }, 300); // Debounce to avoid excessive writes
    
    return () => clearTimeout(timeout);
  }, [state, sliceKey, isHydrated]);

  // Listen for cross-tab updates
  useEffect(() => {
    if (typeof BroadcastChannel === 'undefined') return;
    
    const channel = new BroadcastChannel('fieldkit_sync');
    channel.onmessage = (event) => {
      if (event.data.sliceKey === sliceKey && event.data.type === 'SLICE_UPDATED') {
        // Reload from IndexedDB to sync
        get<T>(sliceKey).then((stored) => {
          if (stored) setState(stored);
        });
      }
    };
    
    return () => channel.close();
  }, [sliceKey]);

  return { state, setState, isHydrated };
};
```

> **Critical**: This hook keeps reducers pure. Persistence and cross-tab sync happen in `useEffect`, not inside state update functions.

```typescript
import { createContext, useContext, useEffect, useReducer } from 'react';

// Existing reducer (keep pure)
const draftReducer = (state: DraftState, action: DraftAction): DraftState => {
  // Pure transformation logic only - NO side effects
  switch (action.type) {

// Provider with persistence side effects (in useEffect)
export const DraftProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [state, dispatch] = useReducer(draftReducer, initialDraftState);
  const [isHydrated, setIsHydrated] = useState(false);

  // Hydrate from IndexedDB on mount
  useEffect(() => {
    let mounted = true;
    get<DraftState>('fieldkit:draft').then((stored) => {
      if (mounted && stored) {
        dispatch({ type: 'HYDRATE', payload: stored });
      }
      setIsHydrated(true);
    });
    return () => { mounted = false; };
  }, []);

  // Persist when state changes (debounced)
  useEffect(() => {
    if (!isHydrated) return;
    
    const timeout = setTimeout(() => {
      set('fieldkit:draft', state).catch(console.error);
      
      // Broadcast to other tabs
      if (typeof BroadcastChannel !== 'undefined') {
        const channel = new BroadcastChannel('fieldkit_sync');
        channel.postMessage({ type: 'DRAFT_UPDATED' });
        channel.close();
      }
    }, 300);
    
    return () => clearTimeout(timeout);
  }, [state, isHydrated]);

  // Listen for cross-tab updates
  useEffect(() => {
    if (typeof BroadcastChannel === 'undefined') return;
    
    const channel = new BroadcastChannel('fieldkit_sync');
    channel.onmessage = (event) => {
      if (event.data.type === 'DRAFT_UPDATED') {
        get<DraftState>('fieldkit:draft').then((stored) => {
          if (stored) dispatch({ type: 'HYDRATE', payload: stored });
        });
      }
    };
    return () => channel.close();
  }, []);

  return (
    <DraftContext.Provider value={{ state, dispatch, isHydrated }}>
      {children}
    </DraftContext.Provider>
  );
};
```

### 4.2 Claim Scanner — EXIF Correction Applied

#### Corrected Camera Hook (No EXIF on Video)
```typescript
// hooks/useCamera.ts - Fixed version
export const useCamera = () => {
  const videoRef = useRef<HTMLVideoElement>(null);
  const [stream, setStream] = useState<MediaStream | null>(null);
  const [error, setError] = useState<string | null>(null);
  const [permission, setPermission] = useState<'prompt' | 'granted' | 'denied'>('prompt');
  
  const startCamera = async () => {
    try {
      // Check permissions
      if (navigator.permissions) {
        const status = await navigator.permissions.query({ name: 'camera' as PermissionName });
        setPermission(status.state as 'prompt' | 'granted' | 'denied');
        if (status.state === 'denied') {
          setError('Camera access denied. Enable in browser settings.');
          return;
        }
      }
      
      const mediaStream = await navigator.mediaDevices.getUserMedia({
        video: { 
          facingMode: 'environment',
          width: { ideal: 1920 },
          height: { ideal: 1080 }
        }
      });
      
      if (videoRef.current) {
        videoRef.current.srcObject = mediaStream;
        // NO EXIF extraction from video element - it doesn't exist on MediaStream
      }
      setStream(mediaStream);
      setError(null);
      setPermission('granted');
    } catch (err: any) {
      setError(err.name === 'NotAllowedError' 
        ? 'Camera access denied.'
        : err.name === 'NotFoundError'
        ? 'No camera found.'
        : 'Camera unavailable.');
      setPermission('denied');
    }
  };
  
  const capture = async () => {
    if (!videoRef.current) return null;
    
    const video = videoRef.current;
    const canvas = document.createElement('canvas');
    
    // Use video dimensions (OS handles orientation for live stream)
    canvas.width = video.videoWidth;
    canvas.height = video.videoHeight;
    
    const ctx = canvas.getContext('2d');
    if (!ctx) return null;
    
    ctx.drawImage(video, 0, 0);
    
    // Convert to Blob with compression
    return new Promise<Blob | null>((resolve) => {
      canvas.toBlob(
        (blob) => {
          if (blob) {
            // Optional: further compress if >2MB using browser-image-compression
            // This library handles EXIF auto-rotation for static files
            if (blob.size > 2 * 1024 * 1024) {
              import('browser-image-compression').then(({ default: compress }) => {
                compress(blob, { maxSizeMB: 1.5, maxWidthOrHeight: 1920 })
                  .then(resolve)
                  .catch(() => resolve(blob));
              });
            } else {
              resolve(blob);
            }
          } else {
            resolve(null);
          }
        },
        'image/jpeg',
        0.85
      );
    });
  };
  
  const stopCamera = () => {
    stream?.getTracks().forEach(track => track.stop());
    setStream(null);
  };
  
  useEffect(() => {
    return () => stopCamera();
  }, []);
  
  return { videoRef, startCamera, capture, stopCamera, error, permission };
};
```

```typescript
export const createFollowUpCalendarEvent = (reminder: Reminder) => {
  const icsContent = [
    'BEGIN:VCALENDAR',
    'VERSION:2.0',
    'PRODID:-//FieldKit//EN',
    'BEGIN:VEVENT',
    `UID:${reminder.id}@fieldkit`,

    'DESCRIPTION:Follow-up reminder from Field Toolkit',
    'END:VEVENT',
    'END:VCALENDAR'

  const blob = new Blob([icsContent], { type: 'text/calendar' });
  const url = URL.createObjectURL(blob);
  
  const link = document.createElement('a');
  link.href = url;
  link.download = `followup-${reminder.id}.ics`;
  link.click();
  
  URL.revokeObjectURL(url);

};
```

### Storage Layer — IndexedDB Only for Domain Data
```typescript
// lib/storage.ts - Simplified, correct abstraction
import { get, set, del, entries } from 'idb-keyval';

export const storage = {
  // Domain data: always IndexedDB
  async saveOpportunity(key: string, value: any) {
    return set(`fieldkit:opp:${key}`, value);
  },
  
  async loadOpportunity(key: string) {
    return get<any>(`fieldkit:opp:${key}`);
  },
  
  async deleteOpportunity(key: string) {
    return del(`fieldkit:opp:${key}`);
  },
  
  // UI preferences: localStorage is fine (small, synchronous)
  savePreference(key: string, value: any) {
    localStorage.setItem(`fieldkit:pref:${key}`, JSON.stringify(value));
  },
  
  loadPreference(key: string) {
    const raw = localStorage.getItem(`fieldkit:pref:${key}`);
    return raw ? JSON.parse(raw) : null;
  },
  
  // Bulk operations for export/import
  async exportAll(): Promise<Record<string, any>> {
    const data: Record<string, any> = {};
    for await (const [key, value] of entries()) {
      if (key.startsWith('fieldkit:opp:')) {
        data[key] = value;
      }
    }
    return data;
  },
  
  async importAll(data: Record<string, any>) {
    for (const [key, value] of Object.entries(data)) {
      await set(key, value);
    }
  }
};
```

### Cross-Tab Sync — BroadcastChannel Implementation
```typescript
// hooks/useBroadcastSync.ts
export const useBroadcastSync = (channelName: string, onMessage: (data: any) => void) => {
  useEffect(() => {
    if (typeof BroadcastChannel === 'undefined') {
      console.warn('BroadcastChannel not supported');
      return;
    }
    
    const channel = new BroadcastChannel(channelName);
    channel.onmessage = (event) => onMessage(event.data);
    
    return () => channel.close();
  }, [channelName, onMessage]);
  
  const broadcast = (data: any) => {
    if (typeof BroadcastChannel !== 'undefined') {
      const channel = new BroadcastChannel(channelName);
      channel.postMessage(data);
      channel.close();
    }
  };
  
  return { broadcast };
};
```
